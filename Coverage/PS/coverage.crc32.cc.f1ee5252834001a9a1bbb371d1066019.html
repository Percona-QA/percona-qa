393 /** Computes x^(len*8) modulo CRC32-C polynomial, which is useful, when you need
394 to conceptually append len bytes of zeros to already computed hash.
395 @param[in] len The value of len in the x^(len*8) mod CRC32-C
396 @return the rest of x^(len*8) mod CRC32-C, with the most significant coefficient
397 of the resulting rest - the one which is for x^31 - stored as the most
398 significant bit of the uint32_t (the one at 1U<<31) */
399 constexpr uint32_t compute_x_to_8len(size_t len) {
400 // The x^(len*8) mod CRC32 polynomial depends on len only.
401 uint32_t x_to_8len = 1;
402 // push len bytes worth of zeros
403 for (size_t i = 0; i < len * 8; ++i) {
404 const bool will_wrap = x_to_8len >> 31 & 1;
405 x_to_8len <<= 1;
406 if (will_wrap) x_to_8len ^= CRC32C_POLYNOMIAL;
407 }
408 return x_to_8len;
409 }

411 /** Produces a 64-bit result by moving i-th bit of 32-bit input to the
412 32-i-th position (zeroing the other bits). Please note that in particular this
413 moves 0-th bit to 32-nd, and 31-st bit to 1-st, so the range in which data
414 resides is not only mirrored, but also shifted one bit. Such operation is useful
415 for implementing polynomial multiplication when one of the operands is given in
416 reverse and we need the result reversed, too (as is the case in CRC32-C):
417 rev(w * v) = rev(w)*flip_at_32(v)
418 proof:
419 rev(w * v)[i] = (w * v)[63-i] = sum(0<=j<=31){w[j]*v[63-i-j]} =
420 sum(0<=j<=31){rev(w)[31-j]*v[63-i-j]} =
421 sum(0<=j<=31){rev(w)[31-j]*flip_at_32(v)[32-63+i+j]} =
422 sum(0<=j<=31){rev(w)[31-j]*flip_at_32(v)[i-(j-31)]} =
423 sum(0<=j<=31){rev(w)[j]*flip_at_32(v)[i-j]} =
424 rev(w)*flip_at_32(v)[i]
425 So, for example, if crc32=rev(w) is the variable storing the CRC32-C hash of a
426 buffer, and you want to conceptually append len bytes of zeros to it, then you
427 can precompute v = compute_x_to_8len(len), and are interested in rev(w*v), which
428 you can achieve by crc32 * flip_at_32(compute_x_to_8len(len)).
429 @param[in] w The input 32-bit polynomial
430 @return The polynomial flipped and shifted, so that i-th bit becomes 32-i-th. */
431 constexpr uint64_t flip_at_32(uint32_t w) {
432 uint64_t f{0};
433 for (int i = 0; i < 32; ++i) {
434 if (((w >> i) & 1)) {
435 f ^= uint64_t{1} << (32 - i);
436 }
437 }
438 return f;
439 }
