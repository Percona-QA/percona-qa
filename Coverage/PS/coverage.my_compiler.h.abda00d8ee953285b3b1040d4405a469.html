181 // enable compiler specified 'diagnostic' pragmas.
182 //
183 // 1. clang on windows defines both clang and msvc pragmas and generates the
184 // same warnings
185 // 2. clang defines both __clang__ and __GNUC__, but doesn't support all GCC
186 // warnings with the same name
187 //
188 // +---------------------+
189 // | enabled diagnostics |
190 // +------+-------+------+
191 // | gcc | clang | msvc |
192 // +-------+------+-------+------+
193 // | gcc | x | - | - |
194 // | clang | - | x | x |
195 // | msvc | - | - | x |
196 // +-------+------+-------+------+
197 // ^^^
198 // +----- current compiler
199 //
200 // suppressions that aren't supported by the compiler are disabled to avoid
201 // "unsupported pragmas" warnings:
202 //
203 // @code

206 // @endcode
