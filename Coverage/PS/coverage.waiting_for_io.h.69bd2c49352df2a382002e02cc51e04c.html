27 namespace iface {

29 /**
30 Interface for reporting idle waiting on IO
31
32 In case when a thread can be blocked by read operation
33 the decoder allows to periodical interrupt the waiting
34 to execute in meantime an action.
35 */
36 class Waiting_for_io {
37 public:
38 32836 virtual ~Waiting_for_io() = default;
39
40 /**
41 Check if idle processing is needed
42
43 This method returns `true` in case when there is
44 an task that needs periodic checking and an IO operation
45 may block the thread for a longer. Such case the IO waiting
46 must be broken for multiple shorter periods.
47 Between those shorter periods the code that does IO must
48 call `on_idle_or_before_read`.
49 */
50 virtual bool has_to_report_idle_waiting() = 0;
51
52 /**
53 Performs idle action
54
55 Method which must implement an action which needs to be
56 executed periodically (asynchronous to any flow). The code
57 that does the IO must call it when `has_to_report_idle_waiting`
58 returns true and one of following occurred:
59
60 * IO code read packet header
61 * IO code is waiting for a header
62 * long-executing-sql
63
64 @retval true OK
65 @retval false I/O error
66 */
67 virtual bool on_idle_or_before_read() = 0;
68 };
