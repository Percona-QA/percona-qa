581 // Analyze the performance of a db by providing cumulative stats over time.
582 // Usage:
583 // Options options;
584 // options.statistics = ROCKSDB_NAMESPACE::CreateDBStatistics();
585 // Status s = DB::Open(options, kDBPath, &db);
586 // ...
587 // options.statistics->getTickerCount(NUMBER_BLOCK_COMPRESSED);
588 // HistogramData hist;
589 // options.statistics->histogramData(FLUSH_TIME, &hist);
590 //
591 // Exceptions MUST NOT propagate out of overridden functions into RocksDB,
