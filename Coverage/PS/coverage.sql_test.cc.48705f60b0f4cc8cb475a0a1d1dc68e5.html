307 /*
308 A fixed-size FIFO pointer queue that also doesn't allow one to put an
309 element that has previously been put into it.
310
311 There is a hard-coded limit of the total number of queue put operations.
312 The implementation is trivial and is intended for use in debug dumps only.
313 */
314
315 template <class T>
316 class Unique_fifo_queue {
317 public:
318 /* Add an element to the queue */
319 void push_back(T *tbl) {
320 if (!tbl) return;
321 // check if we've already scheduled and/or dumped the element
322 for (int i = 0; i < last; i++) {
323 if (elems[i] == tbl) return;
324 }
325 elems[last++] = tbl;
326 }
327
328 bool pop_first(T **elem) {
329 if (first < last) {
330 *elem = elems[first++];
331 return true;
332 }
333 return false;
334 }
335
336 void reset() { first = last = 0; }
337 enum { MAX_ELEMS = 1000 };
338 T *elems[MAX_ELEMS];
339 int first; // First undumped table
340 int last; // Last undumped element
341 };
