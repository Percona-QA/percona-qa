39 /*
40 This structure specifies the maximum amount of resources which
41 can be consumed by each account. Zero value of a member means
42 there is no limit.
43 */
44 typedef struct user_resources {
45 /* Maximum number of queries/statements per hour. */
46 uint questions;
47 /*
48 Maximum number of updating statements per hour (which statements are
49 updating is defined by sql_command_flags array).
50 */
51 uint updates;
52 /* Maximum number of connections established per hour. */
53 uint conn_per_hour;
54 /* Maximum number of concurrent connections. */
55 uint user_conn;
56 /*
57 Values of this enum and specified_limits member are used by the
58 parser to store which user limits were specified in GRANT statement.
59 */
60 enum {
61 QUERIES_PER_HOUR = 1,
62 UPDATES_PER_HOUR = 2,
63 CONNECTIONS_PER_HOUR = 4,
64 USER_CONNECTIONS = 8
65 };
66 uint specified_limits;
67 } USER_RESOURCES;

69 /*
70 This structure is used for counting resources consumed and for checking
71 them against specified user limits.
72 */
73 typedef struct user_conn {
74 /*
75 Pointer to user+host key (pair separated by '\0') defining the entity
76 for which resources are counted (By default it is user account thus
77 priv_user/priv_host pair is used. If --old-style-user-limits option
78 is enabled, resources are counted for each user+host separately).
79 */
80 char *user;
81 /* Pointer to host part of the key. */
82 char *host;
83 /**
84 The moment of time when per hour counters were reset last time
85 (i.e. start of "hour" for conn_per_hour, updates, questions counters).
86 */
87 ulonglong reset_utime;
88 /* Total length of the key. */
89 size_t len;
90 /* Current amount of concurrent connections for this account. */
91 uint connections;
92 /*
93 Current number of connections per hour, number of updating statements
94 per hour and total number of statements per hour for this account.
95 */
96 uint conn_per_hour, updates, questions;
97 /* Maximum amount of resources which account is allowed to consume. */
98 USER_RESOURCES user_resources;
99 } USER_CONN;
