54 using AdjustT = typename std::conditional<
55 std::is_const<typename std::remove_reference<Pattern>::type>::value,
56 typename std::conditional<std::is_lvalue_reference<Pattern>::value,

59 T&&>::type>::type;

62 33 static AdjustT<MockType> Adjust(const MockType& mock) {
63 33 return static_cast<AdjustT<MockType>>(const_cast<MockType&>(mock));

262 // Note: The use of `identity_t` here allows _Ret to represent return types that
263 // would normally need to be specified in a different way. For example, a method
264 // returning a function pointer must be written as
265 //
266 // fn_ptr_return_t (*method(method_args_t...))(fn_ptr_args_t...)
267 //
268 // But we only support placing the return type at the beginning. To handle this,
269 // we wrap all calls in identity_t, so that a declaration will be expanded to
270 //
271 // identity_t<fn_ptr_return_t (*)(fn_ptr_args_t...)> method(method_args_t...)
272 //
273 // This allows us to work around the syntactic oddities of function/method
274 // types.
275 #define GMOCK_INTERNAL_SIGNATURE(_Ret, _Args) \
276 ::testing::internal::identity_t<GMOCK_PP_IF(GMOCK_PP_IS_BEGIN_PARENS(_Ret), \
277 GMOCK_PP_REMOVE_PARENS, \
278 GMOCK_PP_IDENTITY)(_Ret)>( \
279 GMOCK_PP_FOR_EACH(GMOCK_INTERNAL_GET_TYPE, _, _Args))
280
281 #define GMOCK_INTERNAL_GET_TYPE(_i, _, _elem) \
282 GMOCK_PP_COMMA_IF(_i) \
283 GMOCK_PP_IF(GMOCK_PP_IS_BEGIN_PARENS(_elem), GMOCK_PP_REMOVE_PARENS, \
284 GMOCK_PP_IDENTITY) \
285 (_elem)
