65 static void encode_crockford_base32(uint64_t val_to_encode, char *to,
66 size_t encode_len) {
67 assert(encode_len < 14);
68 // Using Douglas Crockford's base 32 encoding to reduce confusion
69 // between similar letters/numbers and avoid accidental obscenities.
70 // See https://www.crockford.com/base32.html
71 constexpr char crockford_base32[] = "0123456789abcdefghjkmnpqrstvwxyz";
72 // Encode just the least significant encode_len * 5 bits of val_to_encode,
73 // starting from the most significant 5 bits
74 const auto BITS_PER_CHAR = 5;
75 assert(encode_len * BITS_PER_CHAR < 64);
76 for (size_t i = encode_len; i--;) {
77 to[i] = crockford_base32[(val_to_encode >> BITS_PER_CHAR * i) % 32];
78 }
79 }

81 /*
82 get_encoded_uuid_excluding_mac returns true on success, false on failure.
83 On success, the to buffer contains 16 chars of an encoded UUID and a null
84 terminator, so the to_len must be at least 17.
85 */
86 static bool get_encoded_uuid_excluding_mac(char *to, size_t to_len) {
87 assert(to_len >= 17);
88 if (to_len < 17) {
89 return false;
90 }
91 UUID uuid;
92 const auto uuid_created = UuidCreateSequential(&uuid);
93 if (!((uuid_created == RPC_S_OK) ||
94 (uuid_created == RPC_S_UUID_LOCAL_ONLY))) {
95 return false;
96 }
97 /*
98 The uuid returned by UuidCreateSequential is a version 1 UUID, as described
99 in rfc4122. The rfc4122 structure maps to the UUID structure thus:
100 unsigned long Data1 is time-low
101 unsigned short Data2 is time-mid
102 unsigned short Data3 is time-high-and-version
103 unsigned char Data4[0] is clock-seq-and-reserved
104 unsigned char Data4[1] is clock-seq-low
105 unsigned char Data4[2..7] is node (MAC address)
106
107 As the MAC address is (usually!) unchanging for a machine, it doesn't
108 help to include it in a "random" temporary filename, so it is not
109 included in the encoded uuid. Including a machine's MAC address in a
110 temporary filename would also leak information unnecessarily.
111 */
112 encode_crockford_base32(
113 uuid.Data1 + (static_cast<uint64_t>(uuid.Data4[0]) << 32), to, 8);
114 encode_crockford_base32(uuid.Data2 +
115 (static_cast<uint64_t>(uuid.Data3) << 16) +
116 (static_cast<uint64_t>(uuid.Data4[1]) << 32),
117 to + 8, 8);
118 to[16] = '\0';
119 return true;
120 }
