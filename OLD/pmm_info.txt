# ======= Server install
  sudo docker create -v /opt/prometheus/data -v /opt/consul-data -v /var/lib/mysql --name pmm-data percona/pmm-server:1.0.5 /bin/true
  docker run -d -p 80:80 --volumes-from pmm-data --name pmm-server --restart always percona/pmm-server:1.0.5
# ======= Server mgmt
  docker exec -it pmm-server bash  # bash client to the docker container
  ps aux | grep '^root' | grep -Evw "$$|([RS]\+)"  # root processes in the docker container
# ======= Client install
  wget https://www.percona.com/redir/downloads/TESTING/pmm/pmm-client-1.0.5-x86_64.tar.gz
  tar -xf pmm-client-1.0.5-x86_64.tar.gz && cd pmm-client-1.0.5-x86_64 && sudo ./install
# ======= Client config (or can use `pmm_os_agent` and `pmm_mysql_agent` scripts generated by startup.sh)
  sudo pmm-admin config --server {IP_ADDRESS_OF_DOCKER_PMM_SERVER}
  sudo pmm-admin add mysql {FREE_CHOSEN_NAME} --socket={DIR_TO_SOCKET}/socket.sock --user=root --query-source=perfschema  # PFS is on by default
